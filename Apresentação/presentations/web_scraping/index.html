Python e BeautifulSoup


import requests
from bs4 import BeautifulSoup

request = requests.get('http://www.ufrn.br/')
soup = BeautifulSoup(request.content, 'html.parser')
for link in soup.find_all('a'):
    print(link.get('href'))

Python, Selenium e BeautifulSoup


from bs4 import BeautifulSoup
from selenium import webdriver

driver = webdriver.Firefox()
driver.get('http://www.ufrn.br/')

soup = BeautifulSoup(driver.find_element_by_tag_name('body').get_attribute('innerHTML'), 'html.parser')

for link in soup.find_all('a'):
    print(link.get('href'))
